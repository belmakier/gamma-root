18 double Fac10(int n)
19 {
20 /* implemented from DCOSUBS.f90
21 calculates n!/10^n */
22 double fac = 1.0;
23 if (n == 0) {
24 return 1;
25 } else if (n > 79) {
26 std::cout << "Can't calculate n! for n>79" << std::endl;
27 return -1;
28 } else if (n < 0) {
29 std::cout << "Can't calculate n! for n<0" << std::endl;
30 return -1;
31 }
32 for (int i = 1; i <= n; ++i) {
33 fac = fac * (double)i / 10.0;
34 }
35 return fac;
36 }

238 int wrresult(char *out, float value, float err, int minlen)
239 {
240 int n;
241 char *c;
242
243 /* append nicely-formatted " value(error)" to string out
244 minlen = requested minimum length of out at completion
245 returns new length of out string */
246 /* from radware */
247
248 c = out + strlen(out);
249 if (err < 0.00005f) {
250 sprintf(c, " %.4f", value);
251 c = out + strlen(out);
252 while (*(c - 1) == '0' && *(c - 2) == '0')
253 c--;
254 strcpy(c, "(0)");
255 } else if (err < 0.0025f) {
256 sprintf(c, " %.4f(%.0f)", value, err * 10000.0f);
257 } else if (err < 0.025f) {
258 sprintf(c, " %.3f(%.0f)", value, err * 1000.0f);
259 } else if (err < 0.25f) {
260 sprintf(c, " %.2f(%.0f)", value, err * 100.0f);
261 } else if (err < 2.5f) {
262 sprintf(c, " %.1f(%.0f)", value, err * 10.0f);
263 } else {
264 sprintf(c, " %.0f(%.0f)", value, err);
265 }
266 for (n = strlen(out); n < minlen; n++) {
267 strcat(out, " ");
268 }
269 return n;
270 } /* wrresult */

297 //This computes the Simpson's-rule integral cross an array of values
298 double Simps(double *y, int n, double dx)
299 {
300 double yint = 0;
301 yint = y[0] + y[n];
302 for(int i=1; i<n; i++) yint += y[i] * double(i%2 + 1) * 2.;
303 yint *= dx/3.;
304
305 return yint;
306 }
307 std::complex<double> **Simps(std::complex<double> ***rho, int n, double dx)
308 {
309 std::complex<double> **rhoint = new std::complex<double>*[3];
310 for(int k=0; k<3; k++) rhoint[k] = new std::complex<double>[5];
311 for(int k=0; k<3; k++)
312 {
313 for(int q=0; q<=k*2; q++) {
314 rhoint[k][q] = std::complex<double>(0,0);
315 rhoint[k][q] = rho[0][k][q] + rho[n][k][q];
316 for(int i=1; i<n; i++) {
317 rho[i][k][q] *= double(i%2 + 1) * 2.;
318 rhoint[k][q] += rho[i][k][q];
319 }
320 rhoint[k][q] *= dx/3.;
321 }
322 }
323
324 return rhoint;
325 }
