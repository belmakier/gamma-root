193 double EffCFunc(double *x, double *pars) {
194 //this is basically from radware
195 double x1 = log(x[0]/100.0);
196 double x2 = log(x[0]/1000.0);
197
198 double g = pars[6];
199 double f1 = pars[0] + pars[1]*x1 + pars[2]*x1*x1;
200 double f2 = pars[3] + pars[4]*x2 + pars[5]*x2*x2;
201
202 double abs_scale = pars[7];
203 double f = 0;
204 double nf = 0;
205 double r = 0;
206 if (f1 <= f2 ) {
207 r = f1/f2;
208 f = f1;
209 nf = f2;
210 }
211 else {
212 r = f2/f1;
213 f = f2;
214 nf = f1;
215 }
216
217 if (r <= 1e-6) {
218 return exp(f)*abs_scale;
219 }
220 else {
221 double y = pow(pow(r, g) + 1.0, -1.0/g);
222 return exp(y*f)*abs_scale;
223 }
224 }

244 void DataSet::SetData(TH1D *fHist, GamR::Nucleus::TransitionMap* fTransitionMap, std::map<std::string, std::pair<double, double> > fIntensityMap)
245 {
246 Clear();
247 std::map<std::string, GamR::Nucleus::Transition > *fTransMap = fTransitionMap->GetMap();
248 fNormalisation = 0.0;
249 double max = 0.0;
250 for (auto &intensity : fIntensityMap) {
251 //check if there's a corresponding transition in transitionmap
252 if (fTransMap->find(intensity.first) != fTransMap->end()) {
253 double area = (*fTransMap)[intensity.first].Apply(fHist);
254 double areaerror = (*fTransMap)[intensity.first].ApplyError(fHist);
255 double energy = intensity.second.first;
256 double intens = intensity.second.second;
257 double eff = area/intens;
258 double efferr = areaerror/intens;
259 if (area<=0) {
260 continue;
261 }
262 fKeys.push_back(intensity.first);
263 fPeakAreas.push_back(area);
264 fPeakErrors.push_back(areaerror);
265 fEnergies.push_back(energy);
266 fIntensities.push_back(intens);
267 fEfficiencies.push_back(eff);
268 fEfficiencyErrors.push_back(efferr);
269
270 if (eff > max) {
271 max = eff;
272 }
273 }
274 }
275
276 fNormalisation = max/1000.0;

553 delete graph;

561 void EffFit::Fit(int quiet/*=0*/)
562 {
563 int nDataSets = fDataSets.size();
564 Int_t npars = 8 + nDataSets;
565 auto gMinuit = new TMinuit(npars);
566 //gMinuit->Command("SET PRINT -1");
567
568 GlobalChiSquare func(this);
569 gMinuit->SetFCN(func.minuit);
570
571 Double_t arglist[10];
572 arglist[0] = 1;
573
574 Int_t ierflg = 0; // some error flag ???
575
576 gMinuit->mnexcm("SET ERR", arglist, 1, ierflg);
577 arglist[0] = 2;
578 gMinuit->mnexcm("SET STR", arglist, 1, ierflg);
579
580 Int_t parnum = 0;
581
582 std::vector< std::string > parnames = {"A", "B", "C", "D", "E", "F", "G", "AbsScale"};
583 for ( int i=0; i<8; ++i ) {
584 double parmin, parmax;
585 EffFunc->GetParLimits(i, parmin, parmax);
586 //std::cout << "Setting parameter " << parnum << " : " << parnames[i].c_str() << " " << EffFunc->GetParameter(i) << " " << EffFunc->GetParError(i) << " " << parmin << " " << parmax;
587 if (parmin==parmax) {
588 parmin = EffFunc->GetParameter(i) - 1; //because Minuit doesn't allow equal upper and lower bounds
589 parmax = EffFunc->GetParameter(i) + 1;
590 }
591 gMinuit->mnparm(parnum, parnames[i].c_str(), EffFunc->GetParameter(i), EffFunc->GetParError(i), parmin, parmax, ierflg);
592 ++parnum;
593 }
594
595 for (int i=0; i<8; ++i) {
596 double parmin, parmax;
597 EffFunc->GetParLimits(i, parmin, parmax);
598 if (parmin*parmax != 0 && parmin >= parmax) {
599 gMinuit->FixParameter(i);
600 }
601 }
602
603 for (int i=0; i<nDataSets; ++i) {
604 std::cout << "Initial fNorm " << i << " = " << fDataSets[i].GetNorm() << std::endl;
605 gMinuit->mnparm(parnum, ("N"+std::to_string(i)).c_str(), fDataSets[i].GetNorm(), 0.1*fDataSets[i].GetNorm(), 0.0, 10*fDataSets[i].GetNorm(), ierflg);
606 if (fDataSets[i].GetNData() == 0 || fDataSets[i].GetAbsolute()) {
607 gMinuit->FixParameter(parnum);
608 }
609 ++parnum;
610 }
611
612 gMinuit->FixParameter(8);
613
614 /* MINIMIZE */
615 arglist[0] = 100000; // Max Iterations
616 arglist[1] = 1; // Number of sigma max step
617 gMinuit->mnexcm("MINIGRAD", arglist, 2, ierflg);
618 gMinuit->mnimpr(); // Improve fit by searching for other minima
619 gMinuit->mnexcm("HESSE", arglist, 2, ierflg);
620 gMinuit->mnexcm("MINOS", arglist, 2, ierflg);
621
622 //get parameters + errors and put into EffFunc
623 for (int i=0; i<8; ++i) {
624 double parVal, parErr;
625 gMinuit->GetParameter(i, parVal, parErr);
626 EffFunc->SetParameter(i, parVal);
627 EffFunc->SetParError(i, parErr);
628 if (i==7) {
629 fAbsScale = parVal;
630 }
631 }
632
633 delete gMinuit;
634
635 if (quiet==0) {
636 TCanvas *c1 = new TCanvas("cEff", "Efficiency Curve", 1280, 720);
637 this->Draw(c1);
638 }
639
640 }
