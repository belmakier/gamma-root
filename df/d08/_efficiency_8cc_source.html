198 double EffCFunc(double *x, double *pars) {
199 //this is basically from radware
200 double x1 = log(x[0]/100.0);
201 double x2 = log(x[0]/1000.0);
202
203 double g = pars[6];
204 double f1 = pars[0] + pars[1]*x1 + pars[2]*x1*x1;
205 double f2 = pars[3] + pars[4]*x2 + pars[5]*x2*x2;
206
207 double abs_scale = pars[7];
208 double f = 0;
209 double nf = 0;
210 double r = 0;
211
212 //if both are negative we have a problem, f1^(-g) + f2^(-g) is potentially undefined
213 //--> return zero
214 if (f1 < 0 && f2 < 0) {
215 if (x[0] > 50 && x[0] < 8192) {
216 std::cerr << "Warning! Both parabolic components of efficiency are < 0 for E_gamma = " << x[0] << " keV, consider decreasing normalization for data sets" << std::endl;
217 }
218 return 0;
219 }
220
221 if (f1 <= f2 ) {
222 r = f1/f2;
223 f = f1;
224 nf = f2;
225 }
226 else {
227 r = f2/f1;
228 f = f2;
229 nf = f1;
230 }
231
232 //if one is negative, and the other positive, or both positive but one is much smaller than the other
233 //approximation where only large component matters
234 if (r <= 1e-6) {
235 double retval = exp(f)*abs_scale;
236 /*
237 if (retval > 0.05) {
238 std::cout << "x = " << x[0] << ", f1, f2 = " << f1 << " " << f2 << std::endl;
239 std::cout << "exp(f) = " << exp(f) << ", abs_scale = " << abs_scale << std::endl;
240 std::cout << "eff = " << retval << std::endl;
241 }
242 */
243 return retval;
244 }
245 //both are positive and approximately the same size
246 //full evaluation
247 else {
248 double y = pow(pow(r, g) + 1.0, -1.0/g);
249 double retval = exp(y*f)*abs_scale;
250 /*
251 if (retval > 0.05) {
252 std::cout << "x = " << x[0] << ", f1, f2 = " << f1 << " " << f2 << std::endl;
253 std::cout << "exp(y*f) = " << exp(y*f) << ", abs_scale = " << abs_scale << std::endl;
254 std::cout << "eff = " << retval << std::endl;
255 }
256 */
257 return retval;
258 }

279 void DataSet::SetData(TH1D *fHist, GamR::Nucleus::TransitionMap* fTransitionMap, std::map<std::string, std::pair<double, double> > fIntensityMap)
280 {
281 Clear();
282 std::map<std::string, GamR::Nucleus::Transition > *fTransMap = fTransitionMap->GetMap();
283 fNormalisation = 0.0;
284 double max = 0.0;
285 for (auto &intensity : fIntensityMap) {
286 //check if there's a corresponding transition in transitionmap
287 if (fTransMap->find(intensity.first) != fTransMap->end()) {
288 double area = (*fTransMap)[intensity.first].Apply(fHist);
289 double areaerror = (*fTransMap)[intensity.first].ApplyError(fHist);
290 double energy = intensity.second.first;
291 double intens = intensity.second.second;
292 double eff = area/intens;
293 double efferr = areaerror/intens;
294 if (area<=0) {
295 continue;
296 }
297 fKeys.push_back(intensity.first);
298 fPeakAreas.push_back(area);
299 fPeakErrors.push_back(areaerror);
300 fEnergies.push_back(energy);
301 fIntensities.push_back(intens);
302 fEfficiencies.push_back(eff);
303 fEfficiencyErrors.push_back(efferr);
304
305 if (eff > max) {
306 max = eff;
307 }
308 }
309 }
310
311 fNormalisation = max/1000.0;

568 p->EffFunc->SetParameter(3, par[3]);
569 p->EffFunc->SetParameter(4, par[4]);

575 for (int i=0; i<8+nDataSets; ++i) {
576 std::cout << Form("%6.3f ", par[i]);
577 }
578
579 for (int i=0; i<nDataSets; ++i) {
580 p->fDataSets[i].SetAbsScale(par[7]);
581 p->fDataSets[i].SetNorm(par[8+i]);
582 if (p->fDataSets[i].GetNData() > 0) {
583 TGraphErrors *graph;
584 if (p->EqualWeights) {
585 graph = p->fDataSets[i].GetGraph();
586 }
587 else {
588 graph = p->fDataSets[i].GetGraphErrors();
589 }
590 if (p->fDataSets[i].GetAbsolute()) { //a bit of extra encouragement here, as for absolute normalizations we typically have a few data points with which we should achieve almost perfect agreement
591 chisq += 100.0*graph->Chisquare(p->EffFunc);
592 }
593 else {
594 chisq += graph->Chisquare(p->EffFunc);
595 }
596
597 delete graph;
598 }
599 }
600 std::cout << Form("chisq=%5.1f",chisq);
601 std::cout << std::endl;
602 return chisq;
603 }

957 residuals->SetLineColor(colors[i]);
958 residuals->SetMarkerColor(colors[i]);
959 graph->Add(residuals);
960 }
961 std_dev = std::sqrt(std_dev/(double)nPoints);
962 std::cout << "Spread = " << std_dev << std::endl;
963 canvas->SetLogy(0);
964 graph->Draw("A*");
965 if (xlow!=xhigh) {
966 graph->GetXaxis()->SetLimits(xlow, xhigh);
967 }
968 graph->Draw("A*");
