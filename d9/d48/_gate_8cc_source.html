297 double Gate::GetIntegral(TH1 *hist, std::vector<Gate> back) const
298 {
299 int nData = 0;
300 double sumX = 0;
301 double sumY = 0;
302 double sumX2 = 0;
303 double sumXY = 0;
304 double sumY2 = 0;
305
306 int startGate = GetBinLow(hist);
307 int stopGate = GetBinHigh(hist);
308
309 for (auto &backgate : back) {
310 int startback = backgate.GetBinLow(hist);
311 int stopback = backgate.GetBinHigh(hist);
312 for (int i=startback; i<=stopback; ++i) {
313 nData++;
314 sumX += hist->GetBinCenter(i);
315 sumX2 += hist->GetBinCenter(i)*hist->GetBinCenter(i);
316 sumXY += hist->GetBinCenter(i)*hist->GetBinContent(i);
317 sumY += hist->GetBinContent(i);
318 sumY2 += hist->GetBinContent(i)*hist->GetBinContent(i);
319 }
320 }
321
322 double slope = (nData*sumXY - sumX*sumY)/(nData*sumX2-sumX*sumX);
323
324 double constant = (sumY*sumX2 - sumX*sumXY)/(nData*sumX2 - sumX*sumX);
325
326 double sumBackground = 0;
327 sumX = 0;
328 sumY = 0;
329 sumX2 = 0;
330 sumY2 = 0;
331 sumXY = 0;
332 for ( int i=startGate; i<= stopGate; ++i) {
333 double background = constant + slope*hist->GetBinCenter(i);
334 sumBackground += background;
335 double Y = hist->GetBinContent(i) - background;
336 sumY += Y;
337 sumXY += hist->GetBinCenter(i)*Y;
338 }
339
340 double posX = sumXY/sumY;
341 double area = (double)sumY;
342 double dArea = std::sqrt(area + 2*sumBackground);
343
344 return area;
345 }
