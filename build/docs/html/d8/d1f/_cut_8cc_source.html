179 TH1D *BackgroundSubtractX(TH2 *hist, GamR::TK::Gate peak, std::vector<GamR::TK::Gate > background, const char *name)
180 {
181 if (background.size()==0) { std::cout << "Must select at least one background region" << std::endl; return NULL; }
182 /* applying gates in X direction */
183 TH1D *hTotal = peak.ApplyX(hist, "hTotal");
184 TH1D *hProjX = (TH1D*)hist->ProjectionX();
185 TH1D *hBackground = background[0].ApplyX(hist, "hBackground");
186 int backgroundWidth = background[0].GetBinWidth(hProjX);
187 for (int i=1; i<background.size(); ++i) {
188 TH1D *hBack = background[i].ApplyX(hist, "hBack");
189 hBackground->Add(hBack);
190 backgroundWidth += background[i].GetBinWidth(hProjX);
191 }
192
193 TH1D *hOut = (TH1D*)hTotal->Clone(name);
194 hOut->SetTitle(name);
195
196 /* get bin widths of gates */
197 double scale =
198 (static_cast<double>(peak.GetBinWidth(hProjX)) /
199 static_cast<double>(backgroundWidth));
200
201 hOut->Add(hBackground, -scale);
202
203 //do correct error propagation
204 for (int i=1; i<=hOut->GetNbinsX(); ++i) {
205 double sigT = hTotal->GetBinError(i);
206 double T = hTotal->GetBinContent(i);
207 double sigB = hBackground->GetBinError(i);
208 double B = hBackground->GetBinContent(i);
209 double sigsquared = pow(sigT, 2) + pow(scale*sigB, 2);
210 hOut->SetBinError(i, sqrt(sigsquared));
211 }
212
213 return hOut;
214 }

424 std::shared_ptr<TH2D> BackgroundEstimate(const TH2 *hist, Int_t NiterX, Int_t NiterY, Int_t direction, Int_t filtertype)
425 {
426 Int_t nbinx = hist->GetNbinsX();
427 Int_t nbiny = hist->GetNbinsY();
428
429 Double_t **spectrum = new Double_t *[nbinx];
430 for (int x = 0; x < nbinx; ++x) {
431 spectrum[x] = new Double_t[nbiny];
432 for (int y = 0; y < nbiny; ++y) {
433 spectrum[x][y] = hist->GetBinContent(x + 1, y + 1);
434 }
435 }
436
437 auto analyser = std::make_unique<TSpectrum2>();
438 analyser->Background(spectrum, nbinx, nbiny, NiterX, NiterY, direction, filtertype);
439
440 auto name = std::string(hist->GetName()) + "_bg";
441 auto output = std::shared_ptr<TH2D>(static_cast<TH2D *>(hist->Clone(name.c_str())));
442 output->SetTitle("Background subtracted");
443 for (int x = 0; x < nbinx; ++x) {
444 for (int y = 0; y < nbiny; ++y) {
445 output->SetBinContent(x + 1, y + 1, spectrum[x][y]);
446 }
447 }
448 delete[] spectrum;
449 return output;
450 }

452 TH2D *BackgroundSubtract2D(TH2 *peak, TH2 *background, double scale) {
453 TH2D *backsub = (TH2D*)peak->Clone(((std::string)peak->GetName()+"_bs").c_str());
454 backsub->Add(background, -scale);
455 //correct error propagation
456 double lastSigT = 1.0;
457 for (int i=0; i<=backsub->GetNbinsX(); ++i) {
458 for (int j=0; j<=backsub->GetNbinsY(); ++j) {
459 double sigT = peak->GetBinError(i,j);
460 double T = peak->GetBinContent(i,j);
461 double sigB = background->GetBinError(i,j);
462 double B = background->GetBinContent(i,j);
463
464 double sigsquared = pow(sigT, 2) + pow(scale*sigB, 2);
465 backsub->SetBinError(i, j, sqrt(sigsquared));
466 lastSigT = sigT;
467 }
468 }
469
470 return backsub;
471 }
